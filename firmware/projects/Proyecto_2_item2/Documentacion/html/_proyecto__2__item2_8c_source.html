120void mostrarTask()

124 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); /*la tarea espera en este punto hasta recibir la notificacion*/
125 if (on)
126 {
127 if (distancia < 10)
128 {
129 LedsOffAll();
130 }
131 else if ((distancia > 10) & (distancia < 20))
132 {
133 LedOn(LED_1);
134 LedOff(LED_2);
135 LedOff(LED_3);
136 }
137
138 else if ((distancia > 20) & (distancia < 30))
139 {
140 LedOn(LED_1);
141 LedOn(LED_2);
142 LedOff(LED_3);
143 }
144
145 else if (distancia > 30)
146 {
147 LedOn(LED_1);
148 LedOn(LED_2);
149 LedOn(LED_3);
150 }
151
152 if (!hold)
153 {
154 LcdItsE0803Write(distancia);
155 }
156 }
157
