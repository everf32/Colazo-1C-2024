82bool Si7007Init(Si7007_config *pins){
83
84
85 GPIOInit(pins->select, GPIO_OUTPUT);
86 GPIOOn(pins->select); //Lo pongo en 1 para que PWM 2 sea temperatura y PWM 1 humedad.
87
88 temp_config.input = pins->PWM_2;;
89 temp_config.mode = ADC_SINGLE;
90
91 hum_config.input = pins->PWM_1;
92 hum_config.mode = ADC_SINGLE;
93
94 AnalogInputInit(&temp_config);
95 AnalogInputInit(&hum_config);
96 return true;
97}
