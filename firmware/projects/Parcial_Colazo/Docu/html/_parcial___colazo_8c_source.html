168void medir() // tarea para realizar las mediciones de ph y humedad
169{
170 while (true)
171 {
172 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
173 if (On) // si on se encuentra en bajo no se mide
174 {
175 AnalogInputReadSingle(CH1, &ph);
176 humedad = GPIORead(GPIO_HUMEDAD);
177 /*se le asigna un valor a ph entre 0V y 3000mV. Tenemos como dato que el sensor tiene una salida desde
178 0V a 3V para un rango de pH desde 0 a 14. Entonces 0V se corresponde a pH = 0 y 3V se corresponde a un
179 pH=14
180 Por lo tanto lo que se plantea es dividir entre 1000 el valor del voltaje asignado a pH y luego hacer regla de
181 3 para calcular que voltaje se corresponde a un pH de 6 y a un pH de 6.7
182 Calculando me da que un ph de 6 se corresponde con un voltaje de 1.28V y un pH de 6.7 se corresponde
183 con un voltaje de 1.43V*/
184 ph = ph / 1000;
185 }
186 }
187}

192void regar()
193{
194 while (true)
195 {
196 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
197 if (On)
198 {
199 /*Para controlar la bomba de agua*/
200 if (!humedad && (ph < umbral_pH_Superior) && (ph > umbral_pH_Inferior))
201 {
202 /*cuando la salida del sensor de humedad es 1, se detecta que el nivel de humedad es superior al deseado
203 y que el pH esta dentro del rango aceptable apago todos los sensores*/
204 GPIOOff(GPIO_BOMBA_AGUA);
205 GPIOOff(GPIO_BOMBA_PHA);
206 GPIOOff(GPIO_BOMBA_PHB);
207 }
208 else if (humedad && (ph < umbral_pH_Superior) && (ph > umbral_pH_Inferior))
209 {
210 GPIOOn(GPIO_BOMBA_AGUA);
211 GPIOOff(GPIO_BOMBA_PHB);
212 GPIOOff(GPIO_BOMBA_PHA);
213 }
214 else if (!humedad && (ph > umbral_pH_Superior))
215 {
216 GPIOOff(GPIO_BOMBA_AGUA);
217 GPIOOff(GPIO_BOMBA_PHB);
218 GPIOOn(GPIO_BOMBA_PHA);
219 }
220
221 else if ((!humedad) && (ph < umbral_pH_Inferior))
222 {
223 /*cuando la salida del sensor de humedad es 0, se detecta que el nivel de humedad es inferior al deseado*/
224 GPIOOff(GPIO_BOMBA_AGUA);
225 GPIOOn(GPIO_BOMBA_PHB);
226 GPIOOff(GPIO_BOMBA_PHA);
227 }
228 else if ((humedad) && (ph < umbral_pH_Inferior))
229 {
230 GPIOOn(GPIO_BOMBA_AGUA);
231 GPIOOn(GPIO_BOMBA_PHB);
232 GPIOOff(GPIO_BOMBA_PHA);
233 }
234 }
235 else
236 {
237 GPIOOn(GPIO_BOMBA_AGUA);
238 GPIOOn(GPIO_BOMBA_PHB);
239 GPIOOff(GPIO_BOMBA_PHA);
240 }
241 }
242}
